<script>
document.addEventListener('DOMContentLoaded', function() {
  let countdown = 5;
  let redirectActive = true;
  const countdownElement = document.getElementById('countdown');
  const cancelButton = document.getElementById('cancel-redirect');
  const homeLink = document.getElementById('home-link');
  
  function updateCountdown() {
    if (!redirectActive) return;
    
    countdownElement.textContent = countdown;
    
    if (countdown <= 0) {
      window.location.href = '/';
      return;
    }
    
    countdown--;
    setTimeout(updateCountdown, 1000);
  }
  
  updateCountdown();
  
  // Cancel redirect functionality
  cancelButton.addEventListener('click', function() {
    redirectActive = false;
    document.querySelector('.auto-redirect').innerHTML = 
      '<p class="small text-muted"><i class="fas fa-check me-1"></i>Auto-redirect cancelled</p>';
  });
  
  // Also cancel redirect if user clicks on any link
  document.addEventListener('click', function(e) {
    if (e.target.tagName === 'A' || e.target.closest('a')) {
      redirectActive = false;
    }
  });
});
</script>
